public class DIRule {
  private double length, quota;

  public static void main(String[] args) throws Exception {
    new DIRule("config.xml");

  public DIRule(String conf) {
    SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMddHHmmssSSS");
    String s = "bi" + sdf.format(new Date());
    try {
      DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
      DocumentBuilder db = dbf.newDocumentBuilder();
      Document d = db.parse(conf);
      d.normalize();
      d.getDocumentElement().getAttributes().getNamedItem("timestamp").setNodeValue(s);

      NodeList nl = d.getDocumentElement().getElementsByTagName("directory");
      for(int i = 0; i < nl.getLength(); i++) {
        NamedNodeMap nnm = nl.item(i).getAttributes();
        Element e = (Element)(nl.item(i).getChildNodes());
        quota = Double.valueOf(e.getElementsByTagName("quota").item(0).getTextContent());
        e.getElementsByTagName("readonly").item(0).setTextContent(
          readonly(
            nnm.getNamedItem("path").getNodeValue() +
            e.getElementsByTagName("name").item(0).getTextContent()
          )
        );
        e.getElementsByTagName("length").item(0).setTextContent(String.format("%.2f", length));
      }

      TransformerFactory tf = TransformerFactory.newInstance();
      Transformer t = tf.newTransformer();
      t.setOutputProperty(OutputKeys.ENCODING, "utf-8");
      t.transform(new DOMSource(d), new StreamResult(conf));
    } catch(Exception e) {
      e.printStackTrace();
    }
  }

  private double getLength(File dir) throws Exception {
    double d = 0;

    if(dir.exists()) {
      for(File f : dir.listFiles()) {
        if(f.isDirectory()) {
          d += getLength(f);
        } else {
          d += f.length();
        }
      }
    }

    return d;
  }

  private String readonly(String pan) throws Exception {
    String s = "icacls \"" + pan + "\" /grant:r everyone:(OI)(CI)";
    if((length = getLength(new File(pan))) > quota * Math.pow(1024, 3)) {
      Runtime.getRuntime().exec(s + "rd");
      s = "Yes";
    } else {
      Runtime.getRuntime().exec(s + "f");
      s = "No";
    }
    
    length /= Math.pow(1024, 3);
    return s;
  }
}
