秒关 - 批处理设定倒计时关机

28 Feb 2019 
@echo off&title SecondsOff - bi20190225080142
set s=%1
 
:main
echo %s%|findstr "^[1-9][0-9]*$ ^0$ ^\-[1-9][0-9]*$"||goto NonNumeric
if %s% lss 0 (shutdown -a) else (shutdown -s -t %s%)
goto %errorlevel%
 
:NonNumeric
cls&set /p s=Please enter a numerical value:
goto main
 
:1190
cls&echo Existing shutdown task!
pause>nul
exit /b
 
:1116
cls&echo No shutdown task!
pause>nul
exit /b
 
:0
cls&echo Set up successfully!
pause>nul
