显示或隐藏指定文件夹的 JScript 脚本

2018-11-07 13:23
var CONST_HIDDEN = 2,
fso = new ActiveXObject("Scripting.FileSystemObject"),
folder = fso.GetFolder("myFolder");
 
if(folder.attributes & CONST_HIDDEN) {
    folder.attributes &= ~CONST_HIDDEN;
} else {
    folder.attributes |= CONST_HIDDEN;
}
