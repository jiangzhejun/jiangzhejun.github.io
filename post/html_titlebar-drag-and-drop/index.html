<script type="text/javascript">
var liulanqi = document.getElementById && !document.all;
document.onmouseup = new Function("shifou=false");

document.onmousedown = function(shijian) {
    var mubiao = liulanqi?shijian.target:event.srcElement;

    while(mubiao.tagName!="HTML" && mubiao.id!="jubing") {
        mubiao = liulanqi?mubiao.parentNode:mubiao.parentElement;
    }

    if(mubiao.id == "jubing") {
        shifou = true;
        hengzuobiao = parseInt(duihuakuang.style.left+0);
        hengyi = liulanqi?shijian.clientX:event.clientX;
        zongzuobiao = parseInt(duihuakuang.style.top+0);
        zongyi = liulanqi?shijian.clientY:event.clientY;
        document.onmousemove = yidong;
        return false;
    }
}

function yidong(shijian) {
    if(shifou) {
        duihuakuang.style.left = (liulanqi?
            hengzuobiao+shijian.clientX-hengyi:
            hengzuobiao+event.clientX-hengyi)+"px";
        duihuakuang.style.top = (liulanqi?
            zongzuobiao+shijian.clientY-zongyi:
            zongzuobiao+event.clientY-zongyi)+"px";
        return false;
    }
}

function zhaohuan(duixiang, zhuangtai) {
    duixiang.style.left = (document.body.clientWidth-
        duixiang.offsetWidth)/2 + "px";
    duixiang.style.top = (document.body.clientHeight-
        duixiang.offsetHeight)/2 + document.body.scrollTop + "px";
    duixiang.style.visibility = zhuangtai;
}
</script>
