@echo off&title %~n0 - bi202009221735

setlocal enabledelayedexpansion
set ipall=ipconfig /all
call:lookup 本地连接 以太网

:main
for /f "delims=. tokens=1,2,3" %%a in ("%~n0") do (
  if "%nic%"=="" nic=%%a
  netsh interface ip set addr "%nic%" static 192.168.%%b.%%c 255.255.255.0 192.168.%%b.1
  netsh interface ip set dns "%nic%" static 211.167.230.100 primary validate=no
  netsh interface ip add dns "%nic%" 211.167.230.200 validate=no
)

cls
%ipall%
pause
exit

:lookup
for %%l in (%*) do (
  for /f "tokens=2*" %%m in ('%ipall%^|find /i "%%l"') do set name=%%m
  for /f "tokens=1* delims=:" %%n in ("!name!") do set nic=%%n
  if "%nic%"=="%%l" (
    goto main
  )
)
