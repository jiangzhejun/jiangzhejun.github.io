<script type="text/javascript">
var nyrmubiao, gbx = gby = 0;
var nyrneishang = document.getElementById("nyrneishang");
var nyrneixia = document.getElementById("nyrneixia");
var nyrwaike = document.getElementById("nyrwaike");
document.onmousemove = gbweizhi;

function gbweizhi(shijian) {
    var shijian= shijian || window.event;

    if(shijian.pageX || shijian.pageY) {
        gbx = shijian.pageX;
        gby = shijian.pageY;
    }
    else {
        gbx = shijian.clientX +
        document.body.scrollLeft - document.body.clientLeft;
        gby = shijian.clientY +
        document.body.scrollTop - document.body.clientTop;
    }
}

function nyrweizhi(mubiao) {
    nyrmubiao = mubiao;
    nyrwaike.style.left = gbx + "px";
    nyrwaike.style.top = (gby + 15) + "px";
    nyrxianshi(-1, -1, 0);
}

function nyrxianshi(nian, yue, qianhou) {
    if(nian < 0) {
        nian = (new Date()).getFullYear();
    }

    if(yue < 0) {
        yue = (new Date()).getMonth();
    }

    yue += qianhou;

    var xingqi, ri;
    var benyue = new Date(nian, yue, 1);
    var xiayue = new Date(nian, yue+1, 1);
    var tianshu = (xiayue - benyue)/1000/60/60/24;

    nian = benyue.getFullYear();
    yue = benyue.getMonth();
    xingqi = benyue.getDay();

    ri = "<table>";
    ri += "<tr>";
    ri += "<td class='xingqi'>";
    ri += "日";
    ri += "</td>";
    ri += "<td class='xingqi'>";
    ri += "一";
    ri += "</td>";
    ri += "<td class='xingqi'>";
    ri += "二";
    ri += "</td>";
    ri += "<td class='xingqi'>";
    ri += "三";
    ri += "</td>";
    ri += "<td class='xingqi'>";
    ri += "四";
    ri += "</td>";
    ri += "<td class='xingqi'>";
    ri += "五";
    ri += "</td>";
    ri += "<td class='xingqi'>";
    ri += "六";
    ri += "</td>";
    ri += "</tr>";
    ri += "<tr>";

    for(i=0; i<xingqi; i++) {
        ri += "<td></td>";
    }

    for(i=0; i<tianshu; i++) {
        ri += "<td onclick='nyrmubiao.value=\"" + nian + "-" + (yue + 1) + "-" + (i + 1) + "\";nyryincang()'>" + (i + 1) + "</td>";

        if((i + xingqi) % 7 == 6) {
            ri += "</tr><tr>";
        }
    }

    ri += "</tr>";
    ri += "</table>";

    nyrneishang.innerHTML = "<span onclick='nyrxianshi(" + nian + ", " + yue + ", -1)' title='上个月'>&lt;</span> <span onclick='nyryincang()' title='点击隐藏'>" + nian + "-" + (yue + 1) + "</span> <span onclick='nyrxianshi(" + nian + ", " + yue + ", 1)' title='下个月'>&gt;</span>";
    nyrneixia.innerHTML = ri;

    nyrwaike.style.visibility = "visible";
}

function nyryincang() {
    nyrwaike.style.visibility = "hidden";
}
</script>
