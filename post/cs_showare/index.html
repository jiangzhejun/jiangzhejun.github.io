    public static Image bianhuan(Bitmap weitu, Boolean shifou, string fangxiang) {
        if(shifou) {
            for(int i=0; i<weitu.Width; i++)
                for(int j=0; j<weitu.Height; j++) {
                    Color caise = weitu.GetPixel(i, j);
                    int xiangsu = (caise.R + caise.G + caise.B)/3;
                    Color huise = Color.FromArgb(xiangsu, xiangsu, xiangsu);
                    weitu.SetPixel(i, j, huise);
                }
        }

        switch(fangxiang) {
        case "左":
            weitu.RotateFlip(RotateFlipType.Rotate270FlipNone);
            break;
        case "右":
            weitu.RotateFlip(RotateFlipType.Rotate90FlipNone);
            break;
        }

        return weitu;
    }

    public static string tianchong(OpenFileDialog duihuakuang) {
        duihuakuang.Filter = "图像 (*.bmp; *.gif; *.jpg; *.png)|*.bmp; *.gif; *.jpg; *.png";
        duihuakuang.Title = "填充";

        if(duihuakuang.ShowDialog() == DialogResult.OK)
            return duihuakuang.FileName;
        else
            return "";
    }

class caidan: MenuStrip {
    public caidan() {
        Dock = DockStyle.Top;
        xiua.heibai.Click += new EventHandler(chaocai);

        ToolStripMenuItem caipu = new ToolStripMenuItem("样式(&M)");
        ToolStripMenuItem[] caixi = {
            new ToolStripMenuItem("相片(&P)"),
            new ToolStripMenuItem("设定(&S)"),
            new ToolStripMenuItem("帮助(&H)")
        };

        caipu.DropDownItems.Add(new ToolStripMenuItem("一寸(&1)", null, chaocai, Keys.F1));
        caipu.DropDownItems.Add(new ToolStripMenuItem("二寸(&2)", null, chaocai, Keys.F2));
        caipu.DropDownItems.Add(new ToolStripMenuItem("六寸(&6)", null, chaocai, Keys.F6));
        caipu.DropDownItems.Add(new ToolStripSeparator());
        caipu.DropDownItems.Add(new ToolStripMenuItem("复合(&9)", null, chaocai, Keys.F9));
        caixi[0].DropDownItems.Add("填充(&C)", null, chaocai);
        caixi[0].DropDownItems.Add("打印(&D)", null, chaocai);
        caixi[0].DropDownItems.Add(new ToolStripSeparator());
        caixi[0].DropDownItems.Add("退出(&X)", null, chaocai);
        caixi[1].DropDownItems.Add(caipu);
        caixi[1].DropDownItems.Add(xiua.heibai);
        caixi[2].DropDownItems.Add("关于(&A)", null, chaocai);

        Items.AddRange(caixi);
    }

    private void chaocai(object duixiang, EventArgs shijian) {
        switch(duixiang.ToString()) {
        case "填充(&C)":
            xiua.mianban.daoru();
            break;
        case "打印(&D)":
            PrintDocument dayin = new PrintDocument();
            dayin.DefaultPageSettings.PaperSize = new PaperSize("A6", 413, 583);
            xiua.fuzhi(dayin.DefaultPageSettings.Bounds);
            dayin.PrintPage += new PrintPageEventHandler(xiua.mianban.daochu);
            dayin.Print();
            break;
        case "退出(&X)":
            Application.Exit();
            break;
        case "一寸(&1)":
            xiua.mianban.gaibian(9);
            break;
        case "二寸(&2)":
            xiua.mianban.gaibian(4);
            break;
        case "六寸(&6)":
            xiua.mianban.gaibian(1);
            break;
        case "复合(&9)":
            xiua.mianban.gaibian(8);
            break;
        case "黑白(&N)":
            xiua.heibai.Checked = !xiua.heibai.Checked;
            xiua.mianban.bianhuan(xiua.heibai.Checked);
            break;
        case "关于(&A)":
            new en(2);
            break;
        }
    }
}

class xiangkuang: Panel {
    public void yangshi(int shuliang) {
        for(int n=0; n<shuliang; n++) {
            Controls.Add(new xiangpian());
            Controls[n].TabStop = xiua.heibai.Checked;
            Controls[n].Text = Text;
        }

        switch(shuliang) {
        case 1:
            Controls[0].Bounds = new Rectangle(0, 0, 300, 420);
            Controls[0].Tag = "中";
            break;
        case 4:
            for(int n=0; n<shuliang; n++) {
                Controls[n].ClientSize = new Size(100, 140);
                Controls[n].Tag = "中";
            }
            Controls[0].Location = new Point(Width/9, Height/9);
            Controls[1].Location = new Point(Width*5/9, Height/9);
            Controls[2].Location = new Point(Width/9, Height*5/9);
            Controls[3].Location = new Point(Width*5/9, Height*5/9);
            break;
        case 8:
            for(int n=0; n<shuliang; n++) {
                if(n < 2) {
                    Controls[n].ClientSize = new Size(100, 140);
                    Controls[n].Tag = "中";
                } else {
                    Controls[n].ClientSize = new Size(100, 72);
                    if(n%2 < 1)
                        Controls[n].Tag = "左";
                    else
                        Controls[n].Tag = "右";
                }
            }
            Controls[0].Location = new Point(Width/9, Height*5/148);
            Controls[1].Location = new Point(Width*5/9, Height*5/148);
            Controls[2].Location = new Point(Width/9, Height*59/148);
            Controls[3].Location = new Point(Width*5/9, Height*59/148);
            Controls[4].Location = new Point(Width/9, Height*89/148);
            Controls[5].Location = new Point(Width*5/9, Height*89/148);
            Controls[6].Location = new Point(Width/9, Height*119/148);
            Controls[7].Location = new Point(Width*5/9, Height*119/148);
            break;
        case 9:
            for(int n=0; n<shuliang; n++) {
                Controls[n].ClientSize = new Size(72, 100);
                Controls[n].Tag = "中";
            }
            Controls[0].Location = new Point(Width/14, Height/14);
            Controls[1].Location = new Point(Width*8/21, Height/14);
            Controls[2].Location = new Point(Width*29/42, Height/14);
            Controls[3].Location = new Point(Width/14, Height*8/21);
            Controls[4].Location = new Point(Width*8/21, Height*8/21);
            Controls[5].Location = new Point(Width*29/42, Height*8/21);
            Controls[6].Location = new Point(Width/14, Height*29/42);
            Controls[7].Location = new Point(Width*8/21, Height*29/42);
            Controls[8].Location = new Point(Width*29/42, Height*29/42);
            break;
        }
    }

    public void gaibian(int shuliang) {
        if(Controls.Count != shuliang) {
            Controls.Clear();
            yangshi(shuliang);
        }
    }

    public void bianhuan(bool shifou) {
        for(int n=0; n<Controls.Count; n++) {
            Controls[n].TabStop = shifou;
            Controls[n].Invalidate();
        }
    }

    public void daoru() {
        string lujing = xiua.tianchong(new OpenFileDialog());
        if(!lujing.Equals("")) {
            Text = lujing;

            for(int n=0; n<Controls.Count; n++) {
                Controls[n].Text = Text;
                Controls[n].Refresh();
            }
        }
    }

    public void daochu(object duixiang, PrintPageEventArgs shijian) {
        try {
            for(int n=0; n<Controls.Count; n++) {
                Bitmap weitu = new Bitmap(Controls[n].Text);

                shijian.Graphics.DrawImage
                (
                    xiua.bianhuan(weitu, Controls[n].TabStop, Controls[n].Tag.ToString()),
                    xiua.daxiao(Controls.Count*10 + n)
                );

                weitu.Dispose();
            }
        } catch(Exception yichang) {
            Console.WriteLine(yichang);
        }
    }
}

class xiangpian: UserControl {
    private ToolStripMenuItem heibai = new ToolStripMenuItem("黑白(&N)");
    private Point[] dingwei = new Point[3];

    public xiangpian() {
        AllowDrop = true;
        DragDrop += new DragEventHandler(tuofang);
        DragEnter += new DragEventHandler(tuojin);

        ContextMenuStrip = new ContextMenuStrip();
        ContextMenuStrip.Items.Add("填充(&C)", null, chaocai);
        ContextMenuStrip.Items.Add(heibai);
        ContextMenuStrip.Opening += new CancelEventHandler(chaocai);

        heibai.Click += new EventHandler(chaocai);
    }

    protected override void OnDoubleClick(EventArgs shijian) {
        daoru();
    }

    protected override void OnPaint(PaintEventArgs shijian) {
        Graphics huabi = shijian.Graphics;
        Pen xian = new Pen(Color.Red, 1);

        try {
            if(File.Exists(Text)) {
                Image tupian = Image.FromFile(Text);
                huabi.DrawImage
                (
                    xiua.bianhuan
                    (
                        new Bitmap(tupian, new Size(Width, Height)),
                        TabStop,
                        Tag.ToString()
                    ),
                    dingwei
                );
            }
        } catch(Exception yichang) {
            Console.WriteLine(yichang);
        }
        finally {
            huabi.DrawRectangle(xian, 0, 0, dingwei[1].X-1, dingwei[2].Y-1);
            huabi.Dispose();
        }
    }

    protected override void OnResize(EventArgs shijian) {
        dingwei[0] = new Point(0, 0);
        dingwei[1] = new Point(Width, 0);
        dingwei[2] = new Point(0, Height);
    }

    private void tuofang(object duixiang, DragEventArgs shijian) {
        Array shuzu = (Array)shijian.Data.GetData(DataFormats.FileDrop);
        Text = shuzu.GetValue(0).ToString();
        Refresh();
    }

    private void tuojin(object duixiang, DragEventArgs shijian) {
        if(shijian.Data.GetDataPresent(DataFormats.FileDrop))
            shijian.Effect = DragDropEffects.Link;
        else
            shijian.Effect = DragDropEffects.None;
    }

    private void chaocai(object duixiang, EventArgs shijian) {
        heibai.Checked = TabStop;

        switch(duixiang.ToString()) {
        case "填充(&C)":
            daoru();
            break;
        case "黑白(&N)":
            TabStop = !heibai.Checked;
            Invalidate();
            break;
        }
    }

    private void daoru() {
        string lujing = xiua.tianchong(new OpenFileDialog());
        if(!lujing.Equals("")) {
            Text = lujing;
            Refresh();
        }
    }
}
