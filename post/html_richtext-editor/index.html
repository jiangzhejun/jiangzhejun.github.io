<iframe id="shuxie"></iframe>

<script type="text/javascript">
var chaowenben = document.getElementById("shuxie").contentWindow;
chaowenben.document.designMode = "on";
chaowenben.document.contentEditable = true;
chaowenben.document.open();
chaowenben.document.write("");
chaowenben.document.close();

function shezhi(zhiling, lujing) {
    chaowenben.focus();
    chaowenben.document.execCommand(zhiling, false, lujing);
}

function zhantie(chuan) {
    if(window.event) {
        chaowenben.document.selection.createRange().pasteHTML(chuan);
    }
    else {
        shezhi("insertHTML", chuan);
    }
}
</script>
