最简单的MFC窗体创建示例

30 Sep 2018 
#include <afxwin.h>

class hwWnd : public CFrameWnd {
public:
	hwWnd(LPCTSTR sTitle) {
		Create(NULL, sTitle);
	}
};

class hwApp : public CWinApp {
public:
	virtual BOOL InitInstance();
};

BOOL hwApp::InitInstance() {
	m_pMainWnd = new hwWnd(_T("HelloWorld!"));
	m_pMainWnd->ShowWindow(m_nCmdShow);
	return TRUE;
}
