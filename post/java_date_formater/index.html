Java兼容各种主流日期格式转换标准日期格式的函数套装

01 Nov 2018 
public static Date getDate(String sFormat, String sDate) {
    SimpleDateFormat sdf = new SimpleDateFormat(sFormat);
    Date d = null;
    
    try {
        sdf.setLenient(false);
        d = sdf.parse(sDate);
    } catch (Exception e) {
        e.printStackTrace();
    }
    
    return d;
}

public static String getDate(String sDate) {
    String[] sa = {
        "yyyy-MM-dd",
        "yyyyMMdd",
        "yyyy.MM.dd",
        "yyyy/MM/dd",
        "yyyy年MM月dd日"
    };
    
    for(String s : sa) {
        Date d = getDate(s, sDate);
        if(d == null) {
            continue;
        } else {
            return new SimpleDateFormat(sa[0]).format(d);
        }
    }
    
    return "";
}
