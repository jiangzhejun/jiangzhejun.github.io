package buid

import (
	"fmt"
	"math/rand"
	"time"
)

func New() string {
	const c = "89ab"
	s := ""
	r := rand.New(rand.NewSource(time.Now().UnixNano()))

	for i := 0; i < 16; i++ {
		j := r.Intn(0x7fff) % 0xff

		switch i {
		case 6:
			s += fmt.Sprintf("4%x", j % 15)
			break
		case 8:
			s += fmt.Sprintf("%s%x", string(c[r.Intn(4) % 4]), j % 15)
			break
		default:
			s += fmt.Sprintf("%02x", j)
		}

		if i < 10 && i % 2 > 0 && i > 1 {
			s += "-"
		}
	}

	return s
}
