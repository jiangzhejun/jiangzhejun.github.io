# coding: utf-8 / 中文地址拆分演示 - bi202207201359

import cpca, sys

if len(sys.argv) > 1:
    df = cpca.transform(sys.argv[1:])
    for row in df.itertuples():
        print(getattr(row, '省'), getattr(row, '市'), getattr(row, '区'), getattr(row, '地址'))
