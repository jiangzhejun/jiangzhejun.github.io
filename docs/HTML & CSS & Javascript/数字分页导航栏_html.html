<script type="text/javascript">
function navibar(page, pages, param) {
  var a, li, ul = document.createElement("ul");
  for (var i = 1; i <= pages; i++) {
    if (i == 2 && page - 6 > 1) {
      i = page - 6;
    } else if (i == page + 6 && page + 6 < pages) {
      i = pages - 1;
    } else {
      li = document.createElement("li");
      a = document.createElement("a");
      if (i == page){
        a.href = "#";
        a.innerText = i;
        li.className = "navi current_page";
      } else {
        a.href = "?page=" + i + param;
        a.innerText = i == 1 || i == pages ? "|" : i;
        li.className = "navi other_page";
      }
      li.appendChild(a);
      ul.appendChild(li);
    }
  }
  return ul;
}

document.body.appendChild(navibar(48, 50, "&keyword="));
</script>
