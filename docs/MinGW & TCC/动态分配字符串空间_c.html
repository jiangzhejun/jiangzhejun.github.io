#include <stdio.h>
#include <stdlib.h>

int length(const char *str) {
  const char *c = str;
  while(*str) {
    str++;
  }

  return str - c;
}

char *copy(char *dest, const char *src) {
  char *c = dest;
  while(*dest++ = *src++);
  *dest = '\0';

  return c;
}

char *join(char *dest, const char *src) {
  char *c = dest;
  dest += length(dest);

  while(*src) {
    *dest++ = *src++;
  }

  *dest = '\0';
  return c;
}

char *getString(char **dest, const char *src) {
  char *c = *dest;
  *dest = (char *)malloc(sizeof(char) * (length(*dest) + length(src) + 1));
  copy(*dest, c);
  join(*dest, src);
  return c;
}

void main() {
  char *c = "12345@678.", *d = "9\0";
  printf("%s\n", c);
  getString(&c, d);
  printf("%s\n", c);
  free(c);
}
