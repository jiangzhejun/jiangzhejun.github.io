isbn.bat

@echo off & cls & title ISBN checkout v2024.12.13 %bww%

setlocal enabledelayedexpansion
set /a pc=10000+%1, tc=10000+%2
set sc=978-7-%pc:~1, 4%-%tc:~1, 4%
set nc=%sc:-=%
set cc=0

for /l %%p in (0, 1, 11) do (
	set /a num=!nc:~%%p, 1!, mod=%%p%%2
	if !mod!==0 (set /a cc+=num) else (set /a cc+=num*3)
)

set /a cc=(10-cc%%10)%%10
echo %sc%-%cc%
endlocal
